Create table T_Session (
	session_id TEXT PRIMARY KEY,
	user_id INTEGER,
	current_start_id INTEGER,
	update_date TEXT NOT NULL,
	create_date TEXT NOT NULL
)

CREATE TABLE T_Start (
	session_id TEXT,
	start_id INTEGER,
	start_date TEXT NOT NULL,
	number_questions INTEGER NOT NULL,
	level_category TEXT NOT NULL,
	level TEXT NOT NULL,
	update_date TEXT NOT NULL,
	create_date TEXT NOT NULL,
	PRIMARY KEY(session_id, start_id)
)

Create table M_Syntax (
	user_id INTEGER,
	syntax_id INTEGER,
	syntax TEXT NOT NULL,
	meaning TEXT NOT NULL,
	studied_date TEXT NOT NULL,
	study_number INTEGER NOT NULL,
	true_number INTEGER NOT NULL,
	false_number INTEGER NOT NULL,
	true_rate REAL NOT NULL,
	review_interval INTEGER NOT NULL,
	next_review_date TEXT NOT NULL,
	update_date TEXT NOT NULL,
	create_date TEXT NOT NULL,
	PRIMARY KEY(user_id, syntax_id)
)

CREATE TABLE T_History (
	session_id TEXT,
	start_id INTEGER,
	row_num INTEGER,
	syntax_id INTEGER NOT NULL,
	japanese_sentence TEXT NOT NULL,
	answer TEXT,
	result INTEGER,
	correct_answer TEXT,
	explanation TEXT,
	update_date TEXT NOT NULL,
	create_date TEXT NOT NULL,
	PRIMARY KEY(session_id, start_id, row_num)
)

CREATE TABLE M_user (
	user_id INTEGER PRIMARY KEY AUTOINCREMENT, 
	user_name TEXT,
	hashed_password TEXT NOT NULL,
	update_date TEXT NOT NULL,
	create_date TEXT NOT NULL
)
